Basic Usage

Setup

.. code-block:: php
    
    require 'vendor/autoload.php';
    
    use Onetoweb\Mirakl\Client;
    
    // param
    $instance = 'instance';
    $apiKey = 'api_key';
    
    // setup client
    $client = new Client($instance, $apiKey);
